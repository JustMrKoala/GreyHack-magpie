UI = {}

UI.c = function(text, hex)
    return "<color=" + hex + ">" + text + "</color>"
end function

UI.bold = function(text)
    return "<b>" + text + "</b>"
end function

UI.header = function(text)
    bar = ""
    i = 0
    while i < 50
        bar = bar + "─"
        i = i + 1
    end while
    return UI.c(bar, C_DARK) + char(10) + UI.c("  " + text, C_PURPLE) + char(10) + UI.c(bar, C_DARK)
end function

UI.section = function(text)
    return UI.c("┌─ ", C_DARK) + UI.c(text, C_YELLOW) + UI.c(" " + "─" * 10, C_DARK)
end function

UI.row = function(label, value, label_color = null, value_color = null)
    if label_color == null then label_color = C_GRAY
    if value_color == null then value_color = C_WHITE
    return UI.c("  " + label + ": ", label_color) + UI.c(value, value_color)
end function

UI.ok = function(text)
    return UI.c("[+] ", C_GREEN) + UI.c(text, C_WHITE)
end function

UI.err = function(text)
    return UI.c("[!] ", C_RED) + UI.c(text, C_WHITE)
end function

UI.warn = function(text)
    return UI.c("[~] ", C_YELLOW) + UI.c(text, C_WHITE)
end function

UI.info = function(text)
    return UI.c("[*] ", C_BLUE) + UI.c(text, C_WHITE)
end function

UI.prompt_line = function(layer, user, pub_ip, path)
    layer_str = UI.c("(" + layer + ")", C_DARK)
    user_str = UI.c(user, C_GREEN)
    at_str = UI.c("@", C_DARK)
    ip_str = UI.c(pub_ip, C_BLUE)
    sep_str = UI.c(":", C_DARK)
    path_str = UI.c(path, C_PURPLE)
    hash_str = UI.c(" # ", C_YELLOW)
    return layer_str + " " + user_str + at_str + ip_str + sep_str + path_str + hash_str
end function

matches_criteria = function(item, key, value, negate = false)
    result = item[key] == value
    if negate then result = not result
    return result
end function

map.first = function(key, value)
    for each in self.indexes
        if matches_criteria(self[each], key, value) then return self[each]
    end for
end function
map.where = function(key, value)
    ret = {}
    for each in self.indexes
        if matches_criteria(self[each], key, value) then ret.push(self[each])
    end for
    return ret
end function
map.wherenot = function(key, value)
    ret = {}
    for each in self.indexes
        if matches_criteria(self[each], key, value, true) then ret.push(self[each])
    end for
    return ret
end function
map.hasMethod = function(method)
    ref = self
    if ref.hasIndex(method) then return true
    while ref.hasIndex("__isa")
        ref = ref["__isa"]
        if ref.hasIndex(method) then return true
    end while
    return false
end function
list.first = function(key, value)
    for each in self
        if typeof(each) == "string" then
            if key == "is" and each == value then return each
            if key == "contains" and each.is_match(value) then return each
            if key == "len" and each.len == value then return each
        else if typeof(each) == "file" then
            if key == "name" and each.name == value then return each
            if key == "namehas" and each.name.is_match(value) then return each
            if key == "path" and each.path == value then return each
            if key == "permissions" and each.permissions == value then return each
            if key == "has_permission" then
                v = value.values
                c = ""
                for p in v
                    if each.has_permission(p) then c = c + p
                end for
                if c == value then return each
            end if
            if key == "size" and each.size == value then return each
        else if typeof(each) == "port" then
            if key == "port_number" and each.port_number == value then return each
            if key == "is_closed" and each.is_closed == value then return each
            if key == "get_lan_ip" and each.get_lan_ip == value then return each
        else if typeof(each) == "map" then
            if each[key] == value then return each
        end if
    end for
end function

MiniClick = {}
MiniClick.commands = {}
MiniClick.aliases = {}

CommandBuilder = function(name, description)
    obj = {}
    obj.name = name
    obj.description = description
    obj.options = {}
    obj.short_aliases = {}
    obj.arguments = []
    obj.group = "misc"

    obj.add_option = function(name, description, default = null, short_name = null, is_flag = false)
        option_data = {
            "name": name,
            "description": description,
            "default": default,
            "short_name": short_name,
            "is_flag": is_flag,
        }
        self.options[name] = option_data
        if short_name != null then self.short_aliases[short_name] = name
        return self
    end function

    obj.add_argument = function(name, description, is_rest = false)
        self.arguments.push({
            "name": name,
            "description": description,
            "is_rest": is_rest,
        })
        return self
    end function

    obj.set_function = function(func)
        self.func = @func
        return self
    end function

    obj.set_group = function(group)
        self.group = group
        return self
    end function

    obj.build = function()
        if MiniClick.commands.hasIndex(self.name) then
            print(UI.warn("Overwriting existing command: " + self.name))
        end if
        MiniClick.commands[self.name] = {
            "name": self.name,
            "description": self.description,
            "options": self.options,
            "short_aliases": self.short_aliases,
            "arguments": self.arguments,
            "func": self["func"],
            "group": self.group,
        }
    end function

    return obj
end function

MiniClick.create_command = function(name, description)
    return CommandBuilder(name, description)
end function

MiniClick.parse_args = function(command, args)
    parsed = {}
    arg_index = 0
    i = 0

    handle_option = function(option_name, is_short)
        if not command.options.hasIndex(option_name) then
            return Error.init("Unknown option: " + option_name)
        end if
        option = command.options[option_name]
        if option.is_flag then
            parsed[option_name] = true
            return 1
        else
            if i + 1 >= args.len then
                return Error.init("Option " + (is_short and "-" or "--") + option_name + " requires a value")
            end if
            parsed[option_name] = args[i + 1]
            return 2
        end if
    end function

    while i < args.len
        arg = args[i]
        if arg[:2] == "--" then
            num_consumed = handle_option(arg[2:], false)
            if typeof(num_consumed) == "Error" then return num_consumed
            i = i + num_consumed
        else if arg[:1] == "-" then
            short_name = arg[1:]
            if not command.short_aliases.hasIndex(short_name) then
                return Error.init("Unknown option: " + short_name)
            end if
            long_name = command.short_aliases[short_name]
            num_consumed = handle_option(long_name, true)
            if typeof(num_consumed) == "Error" then return num_consumed
            i = i + num_consumed
        else
            if arg_index >= command.arguments.len then
                return Error.init("Too many arguments")
            end if
            arg_def = command.arguments[arg_index]
            arg_name = arg_def.name
            if arg_def.is_rest then
                parsed[arg_name] = args[i:]
                arg_index = arg_index + 1
                break
            else
                parsed[arg_name] = arg
                arg_index = arg_index + 1
                i = i + 1
            end if
        end if
    end while

    for option_name in command.options.indexes
        if not parsed.hasIndex(option_name) then
            if command.options[option_name].is_flag then
                parsed[option_name] = false
            else
                parsed[option_name] = command.options[option_name].default
            end if
        end if
    end for

    if arg_index < command.arguments.len then
        return Error.init("Not enough arguments")
    end if

    return parsed
end function

MiniClick.generate_help = function(command_name = null)
    if command_name == null then
        groups = {}
        for name in MiniClick.commands.indexes
            cmd = MiniClick.commands[name]
            g = cmd.group
            if not groups.hasIndex(g) then groups[g] = []
            groups[g].push(cmd)
        end for

        out = UI.header("MAGPIE COMMAND REFERENCE") + char(10) + char(10)

        group_order = ["session", "recon", "exploit", "wifi", "file", "computer", "shell", "util", "misc"]
        group_labels = {
            "session": "Session & Objects",
            "recon": "Recon & Intel",
            "exploit": "Exploit & Attack",
            "wifi": "WiFi",
            "file": "File System",
            "computer": "Computer",
            "shell": "Shell & Connectivity",
            "util": "Utilities",
            "misc": "Misc",
        }

        for g in group_order
            if not groups.hasIndex(g) then continue
            out = out + UI.section(group_labels[g]) + char(10)
            for cmd in groups[g]
                name_col = UI.c(cmd.name, C_GREEN)
                desc_col = UI.c(cmd.description, C_WHITE)
                out = out + "  " + name_col + UI.c(" — ", C_DARK) + desc_col + char(10)
            end for
            out = out + char(10)
        end for

        if MiniClick.aliases.indexes.len > 0 then
            out = out + UI.section("Aliases") + char(10)
            for alias in MiniClick.aliases.indexes
                out = out + "  " + UI.c(alias, C_PINK) + UI.c(" -> ", C_DARK) + UI.c(MiniClick.aliases[alias], C_GREEN) + char(10)
            end for
            out = out + char(10)
        end if

        out = out + UI.c("  Use ", C_GRAY) + UI.c("command --help", C_YELLOW) + UI.c(" for details on any command.", C_GRAY)
        return out
    else
        if MiniClick.aliases.hasIndex(command_name) then
            command_name = MiniClick.aliases[command_name]
        end if
        if not MiniClick.commands.hasIndex(command_name) then
            return Error.init("Unknown command: " + command_name)
        end if
        command = MiniClick.commands[command_name]
        out = UI.header(command_name.upper + " — " + command.description) + char(10) + char(10)

        if command.arguments.len > 0 then
            out = out + UI.section("Arguments") + char(10)
            for arg in command.arguments
                rest_tag = ""
                if arg.is_rest then rest_tag = UI.c(" (rest)", C_GRAY)
                out = out + "  " + UI.c(arg.name, C_ORANGE) + rest_tag + UI.c(" — ", C_DARK) + UI.c(arg.description, C_WHITE) + char(10)
            end for
            out = out + char(10)
        end if

        if command.options.indexes.len > 0 then
            out = out + UI.section("Options") + char(10)
            for option_name in command.options.indexes
                option = command.options[option_name]
                flag_str = "--" + option.name
                if option.short_name != null then flag_str = flag_str + " (-" + option.short_name + ")"
                extra = ""
                if option.is_flag then
                    extra = UI.c(" (flag)", C_GRAY)
                else if option.default != null then
                    extra = UI.c(" (default: " + option.default + ")", C_GRAY)
                end if
                out = out + "  " + UI.c(flag_str, C_BLUE) + extra + UI.c(" — ", C_DARK) + UI.c(option.description, C_WHITE) + char(10)
            end for
        end if

        return out
    end if
end function

MiniClick.run = function(args)
    command_name = args[0]
    if MiniClick.aliases.hasIndex(command_name) then
        command_name = MiniClick.aliases[command_name]
    end if
    if not MiniClick.commands.hasIndex(command_name) then
        return Error.init("Unknown command: " + command_name + "  (type 'help' for a list)")
    end if

    command = MiniClick.commands[command_name]

    for a in args[1:]
        if a == "--help" then return MiniClick.generate_help(command_name)
    end for

    parsed_args = MiniClick.parse_args(command, args[1:])
    if parsed_args == null or typeof(parsed_args) == "Error" then return parsed_args

    if command["func"] == null then
        return Error.init("No function defined for command: " + command_name)
    end if

    result = command.func(parsed_args)
    if result != null then return result
end function

StateManager.crack_best_wifi = function()
    // Real GH wifi API:
    //   computer.wifi_networks("wlan0")  -> list of "BSSID PWR% ESSID" strings
    //   crypto.airmon("start", "wlan0")  -> enable monitor mode
    //   crypto.aireplay(bssid, essid, acks) -> capture packets to file.cap
    //   crypto.aircrack("/home/user/file.cap") -> returns password string
    //   computer.connect_wifi("wlan0", bssid, essid, pass) -> 1 on success

    // Skip if already connected
    if get_shell.host_computer.active_net_card != null then
        print(UI.ok("WiFi already connected — skipping crack."))
        return true
    end if

    crypto = include_lib("/lib/crypto.so")
    if crypto == null then
        print(UI.err("crypto.so not found in /lib — cannot crack WiFi."))
        return false
    end if
    computer = get_shell.host_computer
    device = "wlan0"

    print(UI.info("Enabling monitor mode..."))
    crypto.airmon("start", device)

    print(UI.info("Scanning for WiFi networks..."))
    networks = computer.wifi_networks(device)
    if networks == null or networks.len == 0 then
        print(UI.warn("No WiFi networks in range."))
        crypto.airmon("stop", device)
        return false
    end if

    // Parse "BSSID PWR% ESSID" and pick strongest signal
    best_bssid = ""
    best_essid = ""
    best_pwr = 0
    for entry in networks
        parts = entry.split(" ")
        if parts.len < 3 then continue
        bssid = parts[0]
        pwr = parts[1].remove("%").to_int
        essid = parts[2]
        if typeof(pwr) != "number" then continue
        if pwr > best_pwr then
            best_pwr = pwr
            best_bssid = bssid
            best_essid = essid
        end if
    end for

    if best_bssid == "" then
        print(UI.err("Could not parse any network entries."))
        crypto.airmon("stop", device)
        return false
    end if

    print(UI.info("Best signal: " + best_essid + " (" + best_bssid + ") " + best_pwr + "%"))
    acks = 300000 / best_pwr
    if acks < 7000 then acks = 7000
    print(UI.info("Capturing packets (" + acks + " ACKs)..."))
    aireplay_result = crypto.aireplay(best_bssid, best_essid, acks)
    if typeof(aireplay_result) == "string" then
        print(UI.err("aireplay failed: " + aireplay_result))
        crypto.airmon("stop", device)
        return false
    end if

    cap_path = home_dir + "/file.cap"
    print(UI.info("Cracking " + cap_path + "..."))
    pass = crypto.aircrack(cap_path)
    if pass == null or typeof(pass) != "string" then
        print(UI.err("aircrack failed — not enough packets? Try again."))
        crypto.airmon("stop", device)
        return false
    end if
    print(UI.ok("Password: " + pass))

    print(UI.info("Connecting to " + best_essid + "..."))
    result = computer.connect_wifi(device, best_bssid, best_essid, pass)
    crypto.airmon("stop", device)
    if result == 1 then
        print(UI.ok("Connected to " + best_essid + "."))
        return true
    end if
    print(UI.err("connect_wifi failed: " + result))
    return false
end function

StateManager.first_layer = function()
    print(UI.header("MAGPIE  FIRST RUN  SETUP"))
    self.cob["exploit_store"] = {}
    self.cob["home_path"] = current_path
    self.cob.layer = -1  // stay at -1 until setup fully completes
    self.cob["partial_object"] = {}
    self.cob["stored_objects"] = []
    self.cob["plugin_stores"] = {}

    // Step 1: WiFi first — need connectivity before anything else
    print(UI.info("Getting WiFi connection..."))
    self.crack_best_wifi()

    // Step 2: copy libs directly from /lib/ — already running as root
    print(UI.info("Copying libs from /lib/..."))
    c = get_shell.host_computer
    for lib in ["aptclient.so", "metaxploit.so", "librshell.so", "crypto.so"]
        src_file = c.File("/lib/" + lib)
        if src_file == null then
            print(UI.warn(lib + " not found in /lib — skipping."))
            continue
        end if
        existing = c.File(current_path + "/" + lib)
        if existing != null then existing.delete
        result = src_file.copy(current_path, lib)
        if typeof(result) == "string" then
            print(UI.err("Failed to copy " + lib + ": " + result))
        else
            print(UI.ok(lib + " ready."))
        end if
    end for

    // Step 3: load libs
    self.cob["aptclient"]    = include_lib(current_path + "/aptclient.so")
    self.cob["home_metax"]   = include_lib(current_path + "/metaxploit.so")
    self.cob["remote_metax"] = include_lib(current_path + "/metaxploit.so")
    self.cob["home_crypto"]  = include_lib(current_path + "/crypto.so")

    if self.cob["home_metax"]  == null then exit(UI.err("metaxploit.so failed to load."))
    if self.cob["home_crypto"] == null then exit(UI.err("crypto.so failed to load."))
    print(UI.ok("Libraries ready."))

    // Step 3.5: autodiscover hackstore repos and install missing dependencies
    apt = self.cob["aptclient"]
    if apt == null then
        print(UI.warn("aptclient.so not loaded — skipping hackstore discovery."))
    else
        REQUIRED_DEPS = ["metaxploit", "crypto", "rshell"]

        print(UI.info("Discovering hackstore repositories..."))
        repo_list = apt.show_repos
        if typeof(repo_list) == "string" then
            print(UI.warn("Could not list repos: " + repo_list))
        else
            known_repos = {}
            for repo in repo_list
                known_repos[repo] = true
            end for
            print(UI.ok("Found " + repo_list.len + " repo(s)."))
        end if

        // Scan LAN for any unregistered hackstore nodes on port 27
        print(UI.info("Scanning LAN for hackstore nodes (port 27)..."))
        router = get_router(get_shell.host_computer.public_ip)
        if router != null then
            for lan_ip in router.devices_lan_ip
                for port in router.device_ports(lan_ip)
                    if port.port_number == 27 and not port.is_closed then
                        repo_addr = lan_ip + ":27"
                        if not known_repos.hasIndex(repo_addr) then
                            print(UI.info("Found new hackstore at " + repo_addr + " — adding..."))
                            add_result = apt.add_repo(lan_ip)
                            if typeof(add_result) == "string" then
                                print(UI.warn("Could not add " + repo_addr + ": " + add_result))
                            else
                                known_repos[repo_addr] = true
                                print(UI.ok("Added repo: " + repo_addr))
                            end if
                        end if
                    end if
                end for
            end for
        else
            print(UI.warn("No router accessible — LAN scan skipped."))
        end if

        // Check each required dep; install if missing
        print(UI.info("Checking dependencies..."))
        for dep in REQUIRED_DEPS
            installed = apt.search(dep)
            already_ok = false
            if typeof(installed) != "string" and installed != null then
                for pkg in installed
                    if pkg.is_match(dep) then
                        already_ok = true
                        break
                    end if
                end for
            end if

            if already_ok then
                print(UI.ok(dep + " already installed."))
            else
                print(UI.info("Installing " + dep + "..."))
                result = apt.get(dep)
                if typeof(result) == "string" then
                    print(UI.warn("Failed to install " + dep + ": " + result))
                else
                    print(UI.ok(dep + " installed."))
                end if
            end if
        end for
        print(UI.ok("Dependency check complete."))
    end if

    // Step 4: plugins
    s = get_shell
    c = s.host_computer
    plugin_dir = current_path + "/plugins"
    if not c.File(plugin_dir) then c.create_folder(current_path, "plugins")
    plugin_folder = c.File(plugin_dir)
    if plugin_folder then
        for p_file in plugin_folder.get_files
            if p_file.name.indexOf(".src") == null then continue
            s.build(p_file.path, plugin_dir, false)
            s.launch(plugin_dir + "/" + p_file.name.remove(".src"))
            c.File(plugin_dir + "/" + p_file.name.remove(".src")).delete
        end for
    end if

    self.load_plugins()
    self.set_object(get_shell)
    self.cob["focused_object"].info.user = active_user
    self.cob["home_object"] = self.cob["focused_object"]
    self["relative_object"] = self.cob["focused_object"]
    self.cob.layer = 0  // mark setup as complete
    print(UI.ok("Setup complete."))
end function

StateManager.load_plugins = function()
    for plugin_name in self.cob["plugin_stores"].indexes
        p_data = self.cob["plugin_stores"][plugin_name]
        for cmd in p_data.commands
            builder = MiniClick.create_command(cmd.name, cmd.description)
            if cmd.hasIndex("arguments") then
                for arg in cmd.arguments
                    builder.add_argument(arg.name, arg.description, arg.hasIndex("is_rest") and arg.is_rest)
                end for
            end if
            if cmd.hasIndex("options") then
                for opt in cmd.options.indexes
                    opt_data = cmd.options[opt]
                    builder.add_option(opt_data.name, opt_data.description, opt_data.default, opt_data.short_name, opt_data.is_flag)
                end for
            end if
            builder.set_function(cmd["function"]).build()
        end for
    end for
    print(UI.ok(self.cob["plugin_stores"].len + " plugin(s) loaded."))
end function

ExploitManager.parse_address = function(unparsed)
    segments = unparsed.split("Unsafe check: ")[1:]
    parsed = []
    for exploit in segments
        match = exploit.matches("<b>\S*<\/b>\.")
        parsed.push(match.values.pull[3:match.len-6])
    end for
    return parsed

ExploitManager.verify_tuple = function(tuple)
    if tuple.len != 2 then return false
    if indexOf(["root", "user", "guest", "unknown"], tuple[0]) == null then return false
    if indexOf(["shell", "computer", "file", "number"], tuple[1]) == null then return false
    return true
end function

ExploitManager.save_exploit = function(libname, libversion, address, string, result_tuple)
    if not self.verify_tuple(result_tuple) then return Error.init("Invalid tuple: " + result_tuple)
    if not StateManager.cob["exploit_store"].hasIndex(libname) then StateManager.cob["exploit_store"][libname] = {}
    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then StateManager.cob["exploit_store"][libname][libversion] = {}
    if not StateManager.cob["exploit_store"][libname][libversion].hasIndex(address) then StateManager.cob["exploit_store"][libname][libversion][address] = {}
    StateManager.cob["exploit_store"][libname][libversion][address][string] = result_tuple
end function

    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then return {}

ExploitManager.check_libpath = function(libname, libversion)
    c = StateManager.get_home_computer()
    if typeof(c) == "Error" then return c
    home_path = StateManager.cob["home_path"]
    exploits_path = home_path + "/exploits"
    lib_path = exploits_path + "/" + libname
    version_path = lib_path + "/" + libversion

    check_dir = function(path, name)
        if not c.File(path + "/" + name) then
            c.create_folder(path, name)
            return false
        end if
        return true
    end function

    check_dir(home_path, "exploits")
    check_dir(exploits_path, libname)
    check_dir(lib_path, libversion)
    return c.File(version_path).get_files.len != 0
end function

ExploitManager.check_exploit = function(libname, libversion, address = null)
    if not StateManager.cob["exploit_store"].hasIndex(libname) then return false
    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then return false
    if address != null and not StateManager.cob["exploit_store"][libname][libversion].hasIndex(address) then return false
    return true
end function

ExploitManager.load_exploit = function(libname, libversion)
    c = StateManager.get_home_computer()
    if typeof(c) == "Error" then return c
    self.check_libpath(libname, libversion)
    version_path = StateManager.cob["home_path"] + "/exploits/" + libname + "/" + libversion
    version_dir = c.File(version_path)
    if version_dir == null or version_dir.is_folder == false then return
    address_files = version_dir.get_files
    if address_files == null or address_files.len == 0 then return
    for file in address_files
        address = file.name
        file_content = file.get_content
        if file_content == "" then continue
        lines = file_content.split(char(10))[:-1]
        for exploit in lines
            exploit_data = exploit.split(" ")
            if exploit_data.len < 3 then continue
            string = exploit_data[0]
            user_type = exploit_data[1]
            object_type = exploit_data[2]
            self.save_exploit(libname, libversion, address, string, [user_type, object_type])
        end for
    end for
end function

ExploitManager.store_exploits = function(libname, libversion)
    if not StateManager.cob["exploit_store"].hasIndex(libname) then return
    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then return
    c = StateManager.get_home_computer()

WiFiManager.auto_crack = function(args)
    crypto = include_lib("/lib/crypto.so")
    if crypto == null then return Error.init("crypto.so not in /lib.")
    computer = get_shell.host_computer
    device = "wlan0"

    print(UI.info("Enabling monitor mode..."))
    crypto.airmon("start", device)

    print(UI.info("Scanning for networks..."))
    raw_nets = computer.wifi_networks(device)
    if raw_nets == null or raw_nets.len == 0 then
        crypto.airmon("stop", device)
        return Error.init("No networks found in range.")
    end if

    // Parse entries into structured list
    networks = []
    for entry in raw_nets
        parts = entry.split(" ")
        if parts.len < 3 then continue
        pwr = parts[1].remove("%").to_int
        if typeof(pwr) != "number" then continue
        networks.push({"bssid": parts[0], "pwr": pwr, "essid": parts[2]})
    end for

    if networks.len == 0 then
        crypto.airmon("stop", device)
        return Error.init("Could not parse any network entries.")
    end if

    // Sort by signal strength descending (bubble sort)
    i = 0
    while i < networks.len - 1
        j = 0
        while j < networks.len - 1 - i
            if networks[j]["pwr"] < networks[j+1]["pwr"] then
                tmp = networks[j]
                networks[j] = networks[j+1]
                networks[j+1] = tmp
            end if
            j = j + 1
        end while
        i = i + 1
    end while

    out = UI.header("WIFI SCAN") + char(10) + char(10)
    i = 0
    for net in networks
        out = out + "  " + UI.c(i + 1 + ": ", C_YELLOW) + UI.c(net["essid"], C_GREEN) + "  " + UI.c(net["bssid"], C_BLUE) + "  " + UI.c(net["pwr"] + "%", C_ORANGE) + char(10)
        i = i + 1
    end for
    print(out)

    target = null
    if args["bssid"] != null then
        for net in networks
            if net["bssid"] == args["bssid"] then target = net
        end for
        if target == null then
            crypto.airmon("stop", device)
            return Error.init("BSSID not found: " + args["bssid"])
        end if
    else
        choice = user_input(UI.c("Select number, 'a' for all, 'e' to exit: ", C_YELLOW))
        if choice == "e" then
            crypto.airmon("stop", device)
            return UI.info("Cancelled.")
        end if
        if choice == "a" then
            for net in networks
                pass = self.crack_network(crypto, computer, device, net["bssid"], net["essid"], net["pwr"])
                if pass != null then
                    crypto.airmon("stop", device)
                    return UI.ok("Connected to " + net["essid"] + ".")
                end if
            end for
            crypto.airmon("stop", device)
            return UI.warn("No networks could be cracked.")
        end if
        idx = choice.to_int
        if typeof(idx) != "number" or idx < 1 or idx > networks.len then
            crypto.airmon("stop", device)
            return Error.init("Invalid choice: " + choice)
        end if
        target = networks[idx - 1]
    end if

    pass = self.crack_network(crypto, computer, device, target["bssid"], target["essid"], target["pwr"])
    crypto.airmon("stop", device)
    if pass == null then return Error.init("Could not crack " + target["essid"] + ".")
    return UI.ok("Connected to " + target["essid"] + ".")
end function

FileManager = {}
FileManager.classID = "FileManager"

FileManager.resolve_path = function(file_object, path)
    if file_object == null or typeof(file_object) != "file" then return Error.init("File object not set.")
    if path == null then path = [] else path = path.trim.split("/")
    temp_file = file_object
    if path.len > 0 and path[0] == "" then
        while temp_file.parent != null
            temp_file = temp_file.parent
        end while
        path = path[1:]
    end if
    for step in path
        step = step.trim
        if step == "" then continue
        if step == ".." then
            if temp_file.parent != null then temp_file = temp_file.parent
        else if step == "." then
            continue
        else
            temp_file = temp_file.get_folders.first("name", step)
            if temp_file == null then return Error.init("Folder not found: " + step)
        end if
    end for
    return temp_file
end function

FileManager.resolve_entry = function(f, name)
    if name == "." then return f
    entry = f.get_files.first("name", name)
    if entry != null then return entry
    entry = f.get_folders.first("name", name)
    if entry != null then return entry
    return Error.init("File not found: " + name)
end function

FileCommands.change_perms = function(args)
    f = StateManager.get_focused_file()
    if typeof(f) == "Error" then return f
    filename    = args["filename"]
    permissions = args["permissions"]
    recursive   = args["recursive"]
    if typeof(permissions.to_int) != "number" then return Error.init("Invalid permissions: " + permissions)
    file = FileManager.resolve_entry(f, filename)
    if typeof(file) == "Error" then return file
    result = file.chmod("u-rwx", recursive)
    if result != "" then return Error.init("Error changing permissions: " + result)
    file.chmod("g-rwx", recursive)
    file.chmod("o-rwx", recursive)
    user_octal  = permissions[0].to_int
    group_octal = permissions[1].to_int
    other_octal = permissions[2].to_int
    user_perms  = "u+"
    if bitAnd(user_octal,  4) then user_perms  = user_perms  + "r"
    if bitAnd(user_octal,  2) then user_perms  = user_perms  + "w"
    if bitAnd(user_octal,  1) then user_perms  = user_perms  + "x"
    group_perms = "g+"
    if bitAnd(group_octal, 4) then group_perms = group_perms + "r"
    if bitAnd(group_octal, 2) then group_perms = group_perms + "w"
    if bitAnd(group_octal, 1) then group_perms = group_perms + "x"
    other_perms = "o+"
    if bitAnd(other_octal, 4) then other_perms = other_perms + "r"
    if bitAnd(other_octal, 2) then other_perms = other_perms + "w"
    if bitAnd(other_octal, 1) then other_perms = other_perms + "x"
    file.chmod(user_perms,  recursive)
    file.chmod(group_perms, recursive)
    file.chmod(other_perms, recursive)
    return UI.ok("Permissions changed.")
end function

ShellCommands.drop_shell = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    confirm = user_input(UI.warn("Drop to a raw shell? This will pause magpie. (y/n): "))

        return start_terminal(s)

end function

    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    return build(s, args["source_path"], args["destination_path"], args["can_import"])
end function

ShellCommands.get_file = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    if args["to_relative"] then
        return scp(s, args["source_path"], args["destination_path"], get_shell)
    end if
    return scp(s, args["source_path"], args["destination_path"], StateManager.get_home_shell())
end function

ShellCommands.put_file = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    if args["from_relative"] then
        return scp(get_shell, args["source_path"], args["destination_path"], s)
    end if
    return scp(StateManager.get_home_shell(), args["source_path"], args["destination_path"], s)
end function

ShellCommands.force_read = function(args)
    return UI.warn("force_read is not yet functional.")
end function

ShellCommands.run_command = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    program = args["program"]
    params  = args["params"].join(" ")
    confirm = user_input(UI.warn("Run external program '" + program + "'? (y/n): "))
    if confirm == "y" then return launch(s, program, params)
    return UI.info("Cancelled.")
end function

ShellCommands.connect_service = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    service  = args["service"]
    ip       = args["ip"]
    port     = args["port"]
    user     = args["user"]
    password = args["password"]

    if service == "ssh" then
        shell = s.connect_service(ip, port.to_int, user, password, "ssh")
    else if service == "ftp" then
        shell = s.connect_service(ip, port.to_int, user, password, "ftp")
    else
        return Error.init("Unsupported service: " + service)
    end if

    if typeof(shell) == "string" then return Error.init("Connection failed: " + shell)
    if shell == null then return Error.init("Unknown error connecting to service.")
    result = StateManager.set_object(shell)
    if result == "string" then return Error.init("Error setting object: " + result)
    StateManager.cob["focused_object"]["info"]["user"] = user
    return UI.ok("Connected to " + ip + " via " + service + " as " + user)
end function

ShellCommands.try_ping = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    ip = args["ip"]
    is_pingable = ping(s, ip)
    if typeof(is_pingable) == "string" then return is_pingable
    if is_pingable then return UI.ok("Ping to " + ip + " successful.")
    return UI.warn("Ping to " + ip + " failed.")
end function

ShellCommands.try_sudo = function(args)
    s = StateManager.get_focused_shell()
    if typeof(s) == "Error" then return s
    user     = args["user"]
    password = args["password"]
    result   = get_shell(user, password)
    if result == null or typeof(result) != "shell" then return Error.init("sudo failed: invalid user or password.")
    StateManager.set_object(result)
    StateManager.cob["focused_object"]["info"]["user"] = user
    return UI.ok("Elevated to: " + user)
end function

ShellCommands.pivot_host = function(args)
    s = StateManager.get_focused_shell()
    f = StateManager.get_focused_file()
    if typeof(s) == "Error" then return s
    if typeof(f) == "Error" then return f
    result = scp(StateManager.get_home_shell(), StateManager.cob["home_path"] + "/magpie", f.path, s)
    if typeof(result) == "string" then return Error.init("Error dropping files: " + result)
    result = launch(s, f.path + "/magpie")
    if typeof(result) == "string" then return Error.init("Error launching: " + result)
    return UI.ok("Pivot launched.")
end function

        if StateManager.cob["layer"] == 0 then
            print(UI.info("Saving exploit store..."))
            for lib in StateManager.cob["exploit_store"].indexes

                    if not ExploitManager.check_libpath(lib, version) then
                        ExploitManager.store_exploits(lib, version)
                    end if
                end for
            end for
            print(UI.ok("Exploit store saved. Goodbye."))
        end if
        break
    end if

    tokens = command.split(" ")
    cleaned = []
    for t in tokens
        if t != "" then cleaned.push(t)
    end for

        print(UI.err(result.message))
    else if result != null then
        print(result)

end while
