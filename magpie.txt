matches_criteria = function(item, key, value, negate = false)
    result = item[key] == value
    if negate then result = not result
    return result
end function

map.first = function(key, value)
    for each in self.indexes
        if matches_criteria(self[each], key, value) then return self[each]
    end for
end function
map.where = function(key, value)
    ret = {}
    for each in self.indexes
        if matches_criteria(self[each], key, value) then ret.push(self[each])
    end for
    return ret
end function
map.wherenot = function(key, value)
    ret = {}
    for each in self.indexes
        if matches_criteria(self[each], key, value, true) then ret.push(self[each])
    end for
    return ret
end function
map.hasMethod = function(method)
    ref = self
    if ref.hasIndex(method) then return true
    while ref.hasIndex("__isa")
        ref = ref["__isa"]
        if ref.hasIndex(method) then return true
    end while
    return false
end function
list.first = function(key, value)
    for each in self
        if typeof(each) == "string" then
            if key == "is" and each == value then return each
            if key == "contains" and each.is_match(value) then return each
            if key == "len" and each.len == value then return each
        else if typeof(each) == "file" then
            if key == "name" and each.name == value then return each
            if key == "namehas" and each.name.is_match(value) then return each
            if key == "path" and each.path == value then return each
            if key == "permissions" and each.permissions == value then return each
            if key == "has_permission" then
                v = value.values
                c = ""
                for p in v
                    if each.has_permission(p) then c = c + p
                end for
                if c == value then return each
            end if
            if key == "size" and each.size == value then return each
        else if typeof(each) == "port" then
            if key == "port_number" and each.port_number == value then return each
            if key == "is_closed" and each.is_closed == value then return each
            if key == "get_lan_ip" and each.get_lan_ip == value then return each
        else if typeof(each) == "map" then
            if each[key] == value then return each
        end if
    end for
end function

string.regex_escape = function
    result = self
    for x in "+*?^$.[]{}()|/"
        result = result.replace("\"+x, "\"+x)

Colors.wrap = function(hex, text, bold)
    prefix = "<color=" + hex + ">"
    if bold then prefix = prefix + "<b>"
    suffix = ""
    if bold then suffix = "</b>"
    return prefix + text + suffix + "</color>"
end function

    if Colors.hasIndex(hexc) then hexc = Colors[hexc]
    return Colors.wrap(hexc, self, false)

    if Colors.hasIndex(hexc) then hexc = Colors[hexc]
    return Colors.wrap(hexc, self, true)
end function

// CommandBuilder "Class"
CommandBuilder = function(name, description)
    obj = {}
    obj.name = name
    obj.description = description
    obj.category = "utility"
    obj.options = {}
    obj.short_aliases = {} // short_name -> long_name
    obj.arguments = []

    obj.add_option = function(name, description, default = null, short_name = null, is_flag = false)
        option_data = {
            "name": name,
            "description": description,
            "default": default,
            "short_name": short_name,
            "is_flag": is_flag,
        }
        self.options[name] = option_data
        if short_name != null then
            self.short_aliases[short_name] = name
        end if
        return self // Enable method chaining
    end function

    obj.add_argument = function(name, description, is_rest = false, is_optional = false)
        self.arguments.push({
            "name": name,
            "description": description,
            "is_rest": is_rest,
            "is_optional": is_optional,
        })
        return self
    end function

    obj.set_category = function(category)
        self.category = category
        return self
    end function

    obj.set_function = function(func)
        self.func = @func
        return self
    end function

    obj.build = function()
        if MiniClick.commands.hasIndex(self.name) then
            print("Warning: Overwriting existing command '" + self.name + "'")
        end if
        MiniClick.commands[self.name] = {
            "name": self.name,
            "description": self.description,
            "category": self.category,
            "options": self.options,
            "short_aliases": self.short_aliases,
            "arguments": self.arguments,
            "func": self["func"],
        }
    end function

    return obj
end function

// MiniClick Namespace Methods
MiniClick.create_command = function(name, description)
    return CommandBuilder(name, description)
end function

// Parsing Arguments and Options
MiniClick.parse_args = function(command, args)
    parsed = {}
    arg_index = 0
    i = 0

    handle_option = function(option_name, is_short)
        if not command.options.hasIndex(option_name) then
            return Error.init("Unknown option: " + option_name)
        end if
        option = command.options[option_name]
        if option.is_flag then
            parsed[option_name] = true
            return 1
        else
            if i + 1 >= args.len then
                return Error.init("Option " + (is_short and "-" or "--") + option_name + " requires a value")
            end if
            parsed[option_name] = args[i + 1]
            return 2
        end if
    end function

    while i < args.len
        arg = args[i]
        if arg[:2] == "--" then
            num_consumed = handle_option(arg[2:], false)
            if typeof(num_consumed) == "Error" then return num_consumed
            i = i + num_consumed
        else if arg[:1] == "-" then
            short_name = arg[1:]
            if not command.short_aliases.hasIndex(short_name) then
                return Error.init("Unknown option: " + short_name)
            end if
            long_name = command.short_aliases[short_name]
            num_consumed = handle_option(long_name, true)
            if typeof(num_consumed) == "Error" then return num_consumed
            i = i + num_consumed
        else
            if arg_index >= command.arguments.len then
                return Error.init("Too many arguments")
            end if
            arg_def = command.arguments[arg_index]
            arg_name = arg_def.name
            if arg_def.is_rest then
                parsed[arg_name] = args[i:]
                arg_index = arg_index + 1
                break
            else
                parsed[arg_name] = arg
                arg_index = arg_index + 1
                i = i + 1
            end if
        end if
    end while

    for option_name in command.options.indexes
        if not parsed.hasIndex(option_name) then
            if command.options[option_name].is_flag then
                parsed[option_name] = false
            else
                parsed[option_name] = command.options[option_name].default
            end if
        end if
    end for

    while arg_index < command.arguments.len
        arg_def = command.arguments[arg_index]
        if arg_def.hasIndex("is_optional") and arg_def.is_optional then
            parsed[arg_def.name] = null
            arg_index = arg_index + 1
        else
            return Error.init("Not enough arguments")
        end if
    end while

    return parsed
end function

MiniClick.format_help_rule = function(width)
    rule = ""
    i = 0
    while i < width
        rule = rule + "-"
        i = i + 1
    end while
    return rule.color("dim")
end function

MiniClick.format_help_heading = function(title)
    return char(10) + title.cbold("cyan") + char(10) + MiniClick.format_help_rule(title.len) + char(10)
end function

// Generating Help Text
MiniClick.generate_help = function(command_name = null)
    if command_name == null then
        help_text = char(10) + "Magpie".cbold("gold") + " command reference" + char(10)

        for cat in MiniClick.category_order
            label = MiniClick.category_labels[cat]
            section = ""
            for name in MiniClick.commands.indexes
                command = MiniClick.commands[name]
                cmd_cat = command.hasIndex("category") and command.category or "utility"
                if cmd_cat != cat then continue
                line = "  " + name.cbold("white") + (" - ").color("dim") + command.description.color("grey") + char(10)
                section = section + line
            end for
            if section.len > 0 then
                help_text = help_text + MiniClick.format_help_heading(label) + section
            end if
        end for

        if MiniClick.aliases.indexes.len > 0 then
            help_text = help_text + MiniClick.format_help_heading("Aliases")
            for alias in MiniClick.aliases.indexes
                help_text = help_text + "  " + alias.color("orange") + (" -> ").color("dim") + MiniClick.aliases[alias].cbold("white") + char(10)
            end for
        end if

        help_text = help_text + char(10) + ("Tip: ").color("dim") + "command --help".cbold("cyan") + (" for details.").color("dim")
    else
        if MiniClick.aliases.hasIndex(command_name) then
            command_name = MiniClick.aliases[command_name]
        end if
        if not MiniClick.commands.hasIndex(command_name) then
            return Error.init("Unknown command: " + command_name)
        end if
        command = MiniClick.commands[command_name]
        cmd_cat = command.hasIndex("category") and command.category or "utility"
        cat_label = MiniClick.category_labels.hasIndex(cmd_cat) and MiniClick.category_labels[cmd_cat] or cmd_cat

        help_text = char(10) + command_name.cbold("gold") + (" [" + cat_label + "]").color("dim") + char(10)
        help_text = help_text + command.description.color("white") + char(10)

        if command.arguments.len > 0 then
            help_text = help_text + MiniClick.format_help_heading("Arguments")
            for arg in command.arguments
                req = arg.hasIndex("is_optional") and arg.is_optional and "optional".color("dim") or "required".color("orange")
                rest = arg.hasIndex("is_rest") and arg.is_rest and " rest".color("cyan") or ""
                help_text = help_text + "  " + arg.name.cbold("cyan") + rest + (" (" + req + ")").color("dim") + char(10)
                help_text = help_text + "    " + arg.description.color("grey") + char(10)
            end for
        end if

        if command.options.indexes.len > 0 then
            help_text = help_text + MiniClick.format_help_heading("Options")
            for option_name in command.options.indexes
                option = command.options[option_name]
                opt_line = "  --" + option.name.cbold("green")
                if option.short_name != null then
                    opt_line = opt_line + (" (-" + option.short_name + ")").color("dim")
                end if
                help_text = help_text + opt_line + char(10)
                help_text = help_text + "    " + option.description.color("grey") + char(10)
                if option.is_flag then
                    help_text = help_text + "    " + "(flag)".color("dim") + char(10)
                else if option.default != null then
                    help_text = help_text + "    " + ("default: " + option.default).color("gold") + char(10)
                end if
            end for
        end if
    end if
    return help_text
end function

// Running Commands
MiniClick.run = function(args)
    command_name = args[0]
    if MiniClick.aliases.hasIndex(command_name) then
        command_name = MiniClick.aliases[command_name]
    end if
    if not MiniClick.commands.hasIndex(command_name) then
        return Error.init("Unknown command: " + command_name)
    end if

    command = MiniClick.commands[command_name]

    // --help may appear anywhere in the argument list
    for a in args[1:]
        if a == "--help" then return MiniClick.generate_help(command_name)
    end for

    parsed_args = MiniClick.parse_args(command, args[1:])
    if parsed_args == null or typeof(parsed_args) == "Error" then return parsed_args

    if command["func"] == null then
        return Error.init("Error: No function defined for command '" + command_name + "'")
    end if

    result = command.func(parsed_args)
    if result != null then
        return result
    end if
end function

// Checks random IP addresses for an open port 1542 and returns the IP address.
// @return {string} - The IP address of the hackshop, or null if not found.
get_hackshop = function()
    max_attempts = 50
    attempts = 0
    
    while attempts < max_attempts
        ip = get_random_ip()
        
        if not is_lan_ip(ip) then
            router = get_router(ip)
            
            if router != null then
                ports = router.used_ports
                
                for port in ports
                    if port.port_number == 1542 and not port.is_closed then
                        return router.public_ip
                    end if
                end for
            end if
        end if
        
        attempts = attempts + 1
    end while
    
    return null
end function

WifiManager = {}
WifiManager.CACHE_PATH = "/etc/wifi_cache.txt"
WifiManager.MAX_ACKS = 10000

WifiManager.tag = function(symbol, color_name)
    return ("[" + symbol + "]").cbold(color_name)
end function

WifiManager.get_net_devices = function(computer)
    devices = []
    for line in computer.network_devices.split(char(10))
        line = line.trim
        if line.len == 0 then continue
        parts = line.split(" ")
        cleaned = []
        for p in parts
            if p.len > 0 then cleaned.push(p)
        end for
        if cleaned.len < 3 then continue
        entry = {}
        entry.interface = cleaned[0]
        entry.chipset = cleaned[1]
        entry.monitor_mode = (cleaned[2] == "True")
        devices.push(entry)
    end for
    return devices
end function

WifiManager.get_wlan = function(computer)
    for d in WifiManager.get_net_devices(computer)
        if d.interface.indexOf("wlan") == 0 then return d
    end for
    return null
end function

WifiManager.write_cache = function(computer, bssid, essid, password)
    existing = ""
    f = computer.File(WifiManager.CACHE_PATH)
    if f then existing = f.get_content
    lines = existing.split(char(10))
    kept = []
    for line in lines
        line = line.trim
        if line.len == 0 then continue
        if line.split("|")[0] == bssid then continue
        kept.push(line)
    end for
    kept.push(bssid + "|" + essid + "|" + password)
    computer.touch("/etc", "wifi_cache.txt")
    cf = computer.File(WifiManager.CACHE_PATH)
    if cf then cf.set_content(kept.join(char(10)))
end function

WifiManager.read_cache = function(computer, bssid)
    f = computer.File(WifiManager.CACHE_PATH)
    if not f then return null
    for line in f.get_content.split(char(10))
        line = line.trim
        if line.len == 0 then continue
        parts = line.split("|")
        if parts.len >= 3 and parts[0] == bssid then return parts[2]
    end for
    return null
end function

WifiManager.parse_net = function(net_str)
    parts = net_str.split(" ")
    clean = []
    for p in parts
        if p.len > 0 then clean.push(p)
    end for
    if clean.len < 3 then return null
    result = {}
    result.bssid = clean[0]
    result.signal = clean[1].to_int
    result.essid = clean[2]
    return result
end function

WifiManager.sort_networks = function(nets)
    arr = nets[:]
    n = arr.len
    i = 0
    while i < n - 1
        j = 0
        while j < n - i - 1
            na = WifiManager.parse_net(arr[j])
            nb = WifiManager.parse_net(arr[j + 1])
            sig_a = 0
            sig_b = 0
            if na != null then sig_a = na.signal
            if nb != null then sig_b = nb.signal
            if sig_a < sig_b then
                tmp = arr[j]
                arr[j] = arr[j + 1]
                arr[j + 1] = tmp
            end if
            j = j + 1
        end while
        i = i + 1
    end while
    return arr
end function

WifiManager.try_network = function(computer, crypto, iface, net_str)
    net = WifiManager.parse_net(net_str)
    if net == null then return false

    bssid = net.bssid
    essid = net.essid
    print(WifiManager.tag("*", "cyan") + " Target: " + essid.cbold("white") + " (" + bssid.color("dim") + ") signal: " + str(net.signal).color("gold"))

    cached = WifiManager.read_cache(computer, bssid)
    if cached != null then
        print(WifiManager.tag("*", "cyan") + " Cache hit, trying stored password...")
        if computer.connect_wifi(iface, bssid, essid, cached) then
            print(WifiManager.tag("+", "green") + " Connected via cache: " + essid.cbold("green"))
            return true
        end if
        print(WifiManager.tag("*", "cyan") + " Cached password stale, re-cracking...")
    end if

    cap_file = home_dir + "/file.cap"
    print(WifiManager.tag("*", "cyan") + " Capturing packets (auto-stops at " + str(WifiManager.MAX_ACKS) + " ACKs)...")
    ap_err = crypto.aireplay(bssid, essid, WifiManager.MAX_ACKS)
    if typeof(ap_err) == "string" then
        print(WifiManager.tag("-", "red") + " aireplay error: " + ap_err.color("red"))
        return false
    end if

    cf = computer.File(cap_file)
    if not cf then
        print(WifiManager.tag("-", "red") + " file.cap not found after aireplay.")
        return false
    end if

    print(WifiManager.tag("*", "cyan") + " Cracking " + cap_file.color("gold") + "...")
    password = crypto.aircrack(cap_file)
    if not password then
        print(WifiManager.tag("-", "red") + " aircrack failed for " + essid)
        return false
    end if
    print(WifiManager.tag("+", "green") + " Password: " + password.cbold("green"))

    WifiManager.write_cache(computer, bssid, essid, password)

    print(WifiManager.tag("*", "cyan") + " Connecting to " + essid.cbold("white") + "...")
    if computer.connect_wifi(iface, bssid, essid, password) then
        print(WifiManager.tag("+", "green") + " Connected to " + essid.cbold("green") + "!")
        return true
    end if

    print(WifiManager.tag("-", "red") + " connect_wifi rejected for " + essid)
    return false
end function

WifiManager.ensure_wifi = function()
    computer = get_shell.host_computer
    if computer.is_network_active then return true

    print(WifiManager.tag("*", "orange") + " No network connection detected.".cbold("orange"))
    print(WifiManager.tag("*", "cyan") + " Starting auto WiFi connect...")

    crypto = include_lib("/lib/crypto.so")
    if crypto == null then
        print(WifiManager.tag("-", "red") + " Could not load /lib/crypto.so".cbold("red"))
        return false
    end if

    wlan = WifiManager.get_wlan(computer)
    if wlan == null then
        print(WifiManager.tag("-", "red") + " No wlan interface found.")
        return false
    end if

    iface = wlan.interface
    print(WifiManager.tag("*", "cyan") + " Interface: " + iface.cbold("cyan"))

    if not wlan.monitor_mode then
        print(WifiManager.tag("*", "cyan") + " Starting monitor mode...")
        mon = crypto.airmon("start", iface)
        if typeof(mon) == "string" and mon.lower.indexOf("error") >= 0 then
            print(WifiManager.tag("-", "red") + " airmon failed: " + mon)
            return false
        end if
        print(WifiManager.tag("+", "green") + " Monitor mode on.")
    end if

    print(WifiManager.tag("*", "cyan") + " Scanning for networks...")
    raw = computer.wifi_networks(iface)

    if raw == null or raw.len == 0 then
        crypto.airmon("stop", iface)
        print(WifiManager.tag("-", "red") + " No WiFi networks found.")
        return false
    end if

    sorted = WifiManager.sort_networks(raw)
    print(WifiManager.tag("+", "green") + " Found " + str(sorted.len) + " network(s):")
    for net in sorted
        n = WifiManager.parse_net(net)
        if n != null then
            print("    [" + str(n.signal).color("gold") + "] " + n.essid.cbold("white") + " - " + n.bssid.color("dim"))
        end if
    end for

    max_attempts = sorted.len
    if max_attempts > 2 then max_attempts = 2

    i = 0
    connected = false
    while i < max_attempts and not connected
        if i == 1 then print(WifiManager.tag("*", "orange") + " Primary failed, trying secondary target...")
        connected = WifiManager.try_network(computer, crypto, iface, sorted[i])
        i = i + 1
    end while

    crypto.airmon("stop", iface)

    if connected then
        print(WifiManager.tag("+", "green") + " WiFi connected. You are on the target LAN.".cbold("green"))
        return true

StateManager.load_plugins = function()

            builder = MiniClick.create_command(cmd.name, cmd.description)

            if cmd.hasIndex("arguments") then
                for arg in cmd.arguments
                    builder.add_argument(arg.name, arg.description, arg.hasIndex("is_rest") and arg.is_rest, arg.hasIndex("is_optional") and arg.is_optional)
                end for
            end if

            if cmd.hasIndex("options") then
                for opt in cmd.options.indexes
                    opt_data = cmd.options[opt]
                    builder.add_option(opt_data.name, opt_data.description, opt_data.default, opt_data.short_name, opt_data.is_flag)
                end for
            end if

            if cmd.hasIndex("category") then builder.set_category(cmd.category)
            builder.set_function(cmd["function"]).build()
        end for
    end for

    print(self.cob["plugin_stores"].len + " plugins loaded.")
end function

ExploitManager.parse_payloads = function(scan_raw)
    payloads = []
    segments = scan_raw.split("Unsafe check: ")[1:]

    for segment in segments
        entry = {}
        entry.has_requirement = segment.indexOf("*") != null
        label_start = segment.indexOf("<b>")
        label_end = segment.indexOf("</b>")
        if label_start == null or label_end == null then continue
        entry.name = segment[label_start + 3:label_end]
        payloads.push(entry)
    end for

    return payloads
end function

    if t == "shell" or t == "ftpShell" or t == "computer" or t == "file" or t == "number" then
        tuple = ExploitManager.infer_tuple(result)
        ExploitManager.save_exploit(lib.lib_name, lib.version, address, payload, tuple)
        info.access = tuple[1]
        info.privilege = tuple[0]
        info.usable = true
    end if

    return info
end function

ExploitManager.build_catalog = function(try_all = false, discover = true, address_filter = null, show_progress = false)
    lib = ExploitManager.get_active_lib()
    if typeof(lib) == "Error" then return lib

    metax = ExploitManager.get_metax()
    if typeof(metax) == "Error" then return metax

    libname = lib.lib_name
    libver = lib.version
    catalog = []
    seen = {}
    idx = 1

    saved = ExploitManager.get_exploits(libname, libver)
    for addr in saved.indexes
        if address_filter != null and addr != address_filter then continue
        addr_map = saved[addr]
        for payload in addr_map.indexes
            tuple = addr_map[payload]
            key = addr + "|" + payload
            seen[key] = true
            catalog.push(ExploitManager.catalog_entry(idx, addr, payload, tuple[1], tuple[0], false, null))
            idx = idx + 1
        end for
    end for

    addresses = metax.scan(lib)
    if addresses == null then addresses = []
    StateManager.cob["active_addresses"] = addresses

    for addr in addresses
        if address_filter != null and addr != address_filter then continue
        raw = metax.scan_address(lib, addr)
        if typeof(raw) != "string" then continue

        payloads = ExploitManager.parse_payloads(raw)
        for payload_entry in payloads
            key = addr + "|" + payload_entry.name
            if seen[key] then continue
            seen[key] = true
            catalog.push(ExploitManager.catalog_entry(idx, addr, payload_entry.name, "unknown", "unknown", payload_entry.has_requirement, null))
            idx = idx + 1
        end for
    end for

    if discover then
        if show_progress then
            print(WifiManager.tag("*", "cyan") + " Discovering exploits on " + lib.lib_name.cbold("white") + "...")
        end if

        for entry in catalog
            if entry.access != "unknown" then continue
            if entry.has_requirement and not try_all then continue

            if show_progress then
                print(WifiManager.tag("*", "cyan") + " Testing " + entry.payload.color("white") + "...")
            end if

            test = ExploitManager.test_payload(lib, entry.address, entry.payload)
            if test.usable then
                entry.access = test.access
                entry.privilege = test.privilege
                t = typeof(test.result)
                if t == "shell" or t == "ftpShell" or t == "computer" or t == "file" then
                    entry.stored_result = test.result
                end if
                if show_progress then
                    print(WifiManager.tag("+", "green") + " " + entry.payload.cbold("white") + (" -> ").color("dim") + ExploitManager.format_access_label(test.access))
                end if
            else
                entry.access = "failed"
                if show_progress then
                    print(WifiManager.tag("-", "red") + " " + entry.payload.color("dim") + (" failed").color("red"))
                end if
            end if
        end for
    end if

    StateManager.cob["exploit_catalog"] = catalog
    return catalog
end function

ExploitManager.format_catalog_menu = function(catalog, try_all = false, access_filter = null)
    selectable = []
    for entry in catalog
        if entry.access == "failed" then continue
        if entry.access == "unknown" and entry.has_requirement and not try_all then continue
        if access_filter != null and entry.access != access_filter then continue
        selectable.push(entry)
    end for

    if selectable.len == 0 then
        hint = "No exploits match"
        if access_filter != null then hint = hint + " filter '" + access_filter + "'"
        return Error.init(hint + ". Try scan --all or filter a")
    end if

    n = selectable.len
    i = 0
    while i < n - 1
        j = 0
        while j < n - i - 1
            ka = ExploitManager.access_sort_key(selectable[j].access)
            kb = ExploitManager.access_sort_key(selectable[j + 1].access)
            if ka > kb then
                tmp = selectable[j]
                selectable[j] = selectable[j + 1]
                selectable[j + 1] = tmp
            end if
            j = j + 1
        end while
        i = i + 1
    end while

    lib = StateManager.cob["active_lib"]
    out = char(10) + ("Select exploit").cbold("cyan")
    if lib != null then
        out = out + ("  " + lib.lib_name.color("gold") + " v" + lib.version.color("dim"))
    end if
    out = out + char(10)

    display_idx = 1
    for entry in selectable
        entry.display_index = display_idx
        access_label = ExploitManager.format_access_label(entry.access)
        status = (" ok").color("green")
        if entry.has_requirement then status = (" *").color("orange")
        if entry.access == "unknown" then status = (" ?").color("dim")

        priv = ""
        if entry.privilege != "unknown" and entry.privilege != null then
            priv = (" as ").color("dim") + entry.privilege.color("cyan")
        end if

        line = "  [" + str(display_idx).cbold("gold") + "] " + access_label + priv
        line = line + ("  ").color("dim") + entry.payload.cbold("white") + status + char(10)
        out = out + line
        display_idx = display_idx + 1
    end for

    StateManager.cob["selectable_catalog"] = selectable
    if access_filter != null then
        out = out + char(10) + ("Filter: ").color("dim") + access_filter.cbold("orange") + ("  (").color("dim") + "a".color("cyan") + (" = show all)").color("dim")
    end if
    out = out + char(10) + ExploitManager.format_session_help()
    return out
end function

ExploitManager.format_session_help = function()
    return ("Commands: ").color("dim") + "[#]".cbold("gold") + (" run  ").color("dim") + "r".color("cyan") + (" refresh  ").color("dim") + "f shell|computer|file".color("cyan") + (" filter  ").color("dim") + "a".color("cyan") + (" all  ").color("dim") + "q".color("red") + (" quit").color("dim")
end function

ExploitManager.prompt_opt_if_needed = function(entry)
    if not entry.has_requirement then return ""
    print(("This payload may need an argument (LAN IP, password, path).").color("orange"))
    opt = user_input("Argument (blank to skip): ")
    if opt == null then return ""
    return opt.trim
end function

ExploitManager.run_selection = function(display_index, opt = null)
    if not StateManager.cob.hasIndex("selectable_catalog") then return Error.init("No catalog loaded. Run scan first.")
    selectable = StateManager.cob["selectable_catalog"]

    chosen = null
    for entry in selectable
        if entry.display_index == display_index then chosen = entry
    end for
    if chosen == null then return Error.init("Invalid selection: " + display_index)

    if opt == null and chosen.has_requirement then
        opt = ExploitManager.prompt_opt_if_needed(chosen)
    end if
    if opt == null then opt = ""

    if chosen.stored_result != null then
        return ExploitManager.capture_result(chosen.address, chosen.payload, chosen.stored_result)
    end if

    lib = ExploitManager.get_active_lib()
    if typeof(lib) == "Error" then return lib

    test = ExploitManager.test_payload(lib, chosen.address, chosen.payload, opt)
    if not test.usable then
        out = char(10) + ("Exploit failed.").cbold("red") + char(10)
        out = out + ExploitManager.format_result_type(test.result)
        return out
    end if

    t = typeof(test.result)
    if t == "shell" or t == "ftpShell" or t == "computer" or t == "file" then
        return ExploitManager.capture_result(chosen.address, chosen.payload, test.result)
    end if

    return char(10) + ("Result: ").color("dim") + ExploitManager.format_result_type(test.result)
end function

ExploitManager.interactive_session = function(catalog, try_all, address_filter, default_opt = null, pick = null)
    if pick != null then
        menu = ExploitManager.format_catalog_menu(catalog, try_all, null)
        if typeof(menu) == "Error" then return menu
        pick = pick.to_int
        if typeof(pick) != "number" then return Error.init("Invalid selection: " + pick)
        return ExploitManager.run_selection(pick, default_opt)
    end if

    access_filter = null
    last_result = null

    while true
        menu = ExploitManager.format_catalog_menu(catalog, try_all, access_filter)
        if typeof(menu) == "Error" then return menu

        print(menu)
        choice = user_input("exploit> ")
        if choice == null then return ("Session ended.").color("dim")
        choice = choice.trim
        if choice == "" then continue

        parts = choice.split(" ")
        cmd = parts[0].lower

        if cmd == "q" or cmd == "quit" or cmd == "exit" then
            if last_result != null then return last_result
            return ("Session ended.").color("dim")
        end if

        if cmd == "r" or cmd == "refresh" then
            print(WifiManager.tag("*", "cyan") + " Refreshing exploit list...")
            catalog = ExploitManager.build_catalog(try_all, true, address_filter, true)
            if typeof(catalog) == "Error" then return catalog
            continue
        end if

        if cmd == "a" or cmd == "all" then
            access_filter = null
            continue
        end if

        if cmd == "f" or cmd == "filter" then
            if parts.len < 2 or parts[1].lower == "clear" then
                access_filter = null
                continue
            end if
            access_filter = parts[1].lower
            continue
        end if

        if cmd == "h" or cmd == "help" then
            print(ExploitManager.format_session_help())
            continue
        end if

        pick = cmd.to_int
        if typeof(pick) == "number" then
            result = ExploitManager.run_selection(pick, default_opt)
            print(result)
            last_result = result

            if typeof(result) == "string" and result.indexOf("Exploit succeeded") != null then
                cont = user_input(("Capture another exploit? (y/n): ").color("dim"))
                if cont == null or cont.trim.lower != "y" then return result
            end if
            continue
        end if

        print(("Unknown command: ").color("red") + choice.color("white") + ("  Type h for help.").color("dim"))
    end while
end function

ExploitManager.infer_tuple = function(result)
    t = typeof(result)
    if t == "shell" or t == "ftpShell" then return ["unknown", "shell"]
    if t == "computer" then return ["unknown", "computer"]
    if t == "file" then return ["unknown", "file"]
    if t == "number" then return ["unknown", "number"]
    return null
end function

ExploitManager.format_result_type = function(result)
    t = typeof(result)
    if t == "shell" or t == "ftpShell" then return "shell".cbold("green")
    if t == "computer" then return "computer".cbold("magenta")
    if t == "file" then return "file".color("gold")
    if t == "number" then return ("number: " + str(result)).color("cyan")
    if t == "string" then return result.color("red")
    if t == "null" then return "null (exploit failed)".color("dim")
    return str(result).color("white")
end function

ExploitManager.capture_result = function(address, payload, result)
    t = typeof(result)
    if t != "shell" and t != "ftpShell" and t != "computer" and t != "file" then return null

    lib = StateManager.cob["active_lib"]
    tuple = ExploitManager.infer_tuple(result)
    if tuple != null then
        ExploitManager.save_exploit(lib.lib_name, lib.version, address, payload, tuple)
    end if

    StateManager.set_object(result)
    uid = StateManager.cob["focused_object"]["uid"]

    out = char(10) + ("Exploit succeeded!").cbold("green") + char(10)
    out = out + ("  Type:    ").color("dim") + ExploitManager.format_result_type(result) + char(10)
    out = out + ("  Address: ").color("dim") + address.color("gold") + char(10)
    out = out + ("  Payload: ").color("dim") + payload.color("white") + char(10)
    out = out + ("  UID:     ").color("dim") + uid.cbold("cyan") + ("  (focused)").color("green")
    return out
end function

ExploitManager.auto_exploit = function(try_all = false)
    lib = ExploitManager.get_active_lib()
    if typeof(lib) == "Error" then return lib

    metax = ExploitManager.get_metax()
    if typeof(metax) == "Error" then return metax

    libname = lib.lib_name
    libver = lib.version
    saved = ExploitManager.get_exploits(libname, libver)

    for addr in saved.indexes
        addr_map = saved[addr]
        for payload in addr_map.indexes
            result = lib.overflow(addr, payload)
            captured = ExploitManager.capture_result(addr, payload, result)
            if captured != null then return captured
        end for
    end for

    addresses = metax.scan(lib)

    for addr in addresses
        raw = metax.scan_address(lib, addr)

            result = lib.overflow(addr, entry.name)
            captured = ExploitManager.capture_result(addr, entry.name, result)
            if captured != null then return captured
        end for
    end for

    if not try_all then hint = hint + char(10) + ("Or run ").color("dim") + "auto --all".cbold("orange") + (" to brute-force every payload.").color("dim")

end function

NetworkMapper = {}

NetworkMapper.parse_whois = function(raw)
    info = {}
    for line in raw.split(char(10))
        if line.indexOf(":") == null then continue
        parts = line.split(":")
        key = parts[0].trim.lower
        val = parts[1]
        idx = 2
        while idx < parts.len
            val = val + ":" + parts[idx]
            idx = idx + 1
        end while
        info[key] = val.trim
    end for
    return info
end function

NetworkMapper.get_device_type = function(ip)
    if get_switch(ip) then return "switch"
    if get_router(ip) then return "router"
    return "device"
end function

NetworkMapper.collect_devices = function(router, device_list)
    if router == null then return

    lan_ip = router.local_ip
    if device_list.indexOf(lan_ip) == null then device_list.push(lan_ip)

    for device in router.devices_lan_ip
        if device == lan_ip then continue
        if device_list.indexOf(device) != null then continue

        dtype = NetworkMapper.get_device_type(device)
        device_list.push(device)

        if dtype == "router" then
            NetworkMapper.collect_devices(get_router(device), device_list)
        else if dtype == "switch" then
            NetworkMapper.collect_devices(get_switch(device), device_list)
        end if
    end for
end function

NetworkMapper.build_port_entry = function(router, port, scan_ip)
    entry = {}
    entry.number = port.port_number
    entry.address = port.get_lan_ip
    entry.service = router.port_info(port)

    if port.is_closed then
        entry.status = "closed"
        entry.sort_key = 1
    else if is_lan_ip(scan_ip) then
        entry.status = "open"
        entry.sort_key = 100
    else
        entry.status = "internal"
        entry.sort_key = 10
    end if

    for fwd in router.used_ports
        if fwd.port_number == port.port_number and fwd.get_lan_ip == entry.address and not fwd.is_closed then
            entry.status = "open"
            entry.sort_key = 100
        end if
    end for

    return entry
end function

NetworkMapper.scan = function(target)
    resolved = target
    domain_label = null

    if not is_valid_ip(target) then
        lookup = nslookup(target)
        if lookup == "Not found" then return Error.init("Invalid IP address or domain: " + target)
        domain_label = target
        resolved = lookup
    end if

    router = get_router(resolved)
    if router == null then
        if is_lan_ip(resolved) then
            router = get_router
        else
            return Error.init("No router found for: " + resolved)
        end if
    end if

    device_list = []
    NetworkMapper.collect_devices(router, device_list)

    for port in router.used_ports
        if device_list.indexOf(port.get_lan_ip) == null then device_list.push(port.get_lan_ip)
    end for

    whois_raw = whois(router.public_ip)
    whois_info = NetworkMapper.parse_whois(whois_raw)

    devices = {}
    for device in device_list
        device_info = {}
        device_info.address = device
        device_info.ports = []
        device_info.sort_key = 0
        device_info.is_router = 0

        device_ports = router.device_ports(device)
        if typeof(device_ports) == "string" or device_ports == null then device_ports = []

        for port in router.used_ports
            if port.get_lan_ip == device then
                add_port = true
                for existing in device_ports
                    if existing.port_number == port.port_number and existing.get_lan_ip == device then add_port = false
                end for
                if add_port then device_ports.push(port)
            end if
        end for

        if typeof(device_ports) == "list" then
            for port in device_ports
                port_entry = NetworkMapper.build_port_entry(router, port, resolved)
                device_info.ports.push(port_entry)
                device_info.sort_key = device_info.sort_key + port_entry.sort_key
                if port.port_number == 8080 then device_info.is_router = 1
            end for
        end if

        if device == router.local_ip then
            device_info.ports.push({
                "number": 0,
                "address": device,
                "service": "kernel " + router.kernel_version,
                "status": "open",
                "sort_key": 1000,
            })
            device_info.sort_key = device_info.sort_key + 1000
        end if

        devices[device] = device_info
    end for

    result = {}
    result.target = resolved
    result.domain = domain_label
    result.whois_raw = whois_raw
    result.whois = whois_info
    result.router = router
    result.firewall = router.firewall_rules
    result.devices = devices
    return result
end function

NetworkMapper.format_heading = function(title)
    rule = ""
    i = 0
    while i < title.len
        rule = rule + "-"
        i = i + 1
    end while
    return char(10) + title.cbold("cyan") + char(10) + rule.color("dim") + char(10)
end function

NetworkMapper.format_port_line = function(port_entry)
    status = port_entry.status
    line = "  " + str(port_entry.number) + "  [" + status + "]  " + port_entry.service

    if status == "open" then return line.cbold("green")
    if status == "internal" then return line.color("orange")
    return line.color("grey")
end function

NetworkMapper.format = function(data)
    if data == null then return Error.init("No map data to display.")

    out = char(10) + ("MAP: " + data.target).cbold("gold") + char(10)

    out = out + NetworkMapper.format_heading("WHOIS")
    if data.domain != null then out = out + "  Domain:  " + data.domain.cbold("white") + char(10)
    out = out + "  IP:      " + data.router.public_ip.cbold("green") + char(10)

    if data.whois.hasIndex("domain name") then
        out = out + "  Domain:  " + data.whois["domain name"].color("white") + char(10)
    end if
    if data.whois.hasIndex("administrative contact") then
        out = out + "  Admin:   " + data.whois["administrative contact"].color("white") + char(10)
    end if
    if data.whois.hasIndex("email address") then
        out = out + "  Email:   " + data.whois["email address"].color("blue") + char(10)
    end if
    if data.whois.hasIndex("phone") then
        out = out + "  Phone:   " + data.whois["phone"].color("dim") + char(10)
    end if

    out = out + NetworkMapper.format_heading("ROUTER")
    out = out + "  Public:  " + data.router.public_ip.color("green") + char(10)
    out = out + "  LAN:     " + data.router.local_ip.cbold("gold") + char(10)
    out = out + "  Kernel:  " + data.router.kernel_version.color("cyan") + char(10)
    out = out + "  ESSID:   " + data.router.essid_name.color("white") + char(10)
    out = out + "  BSSID:   " + data.router.bssid_name.color("dim") + char(10)

    if data.firewall and data.firewall.len > 0 then
        out = out + NetworkMapper.format_heading("FIREWALL")
        for rule in data.firewall
            out = out + "  " + rule.color("orange") + char(10)
        end for
    end if

    out = out + NetworkMapper.format_heading("DEVICES")

    sorted_devices = []
    for item in data.devices
        sorted_devices.push(item.value)
    end for

    n = sorted_devices.len
    i = 0
    while i < n - 1
        j = 0
        while j < n - i - 1
            if sorted_devices[j].sort_key < sorted_devices[j + 1].sort_key then
                tmp = sorted_devices[j]
                sorted_devices[j] = sorted_devices[j + 1]
                sorted_devices[j + 1] = tmp
            end if
            j = j + 1
        end while
        i = i + 1
    end while

    for device in sorted_devices
        header = device.address
        if device.address == data.router.local_ip then
            out = out + char(10) + header.cbold("gold") + " " + "(router)".color("gold") + char(10)
        else
            out = out + char(10) + header.cbold("white") + char(10)
        end if

        if device.ports.len == 0 then
            out = out + "  (no ports)".color("dim") + char(10)
            continue
        end if

        port_sorted = device.ports[:]
        pn = port_sorted.len
        pi = 0
        while pi < pn - 1
            pj = 0
            while pj < pn - pi - 1
                if port_sorted[pj].number > port_sorted[pj + 1].number then
                    ptmp = port_sorted[pj]
                    port_sorted[pj] = port_sorted[pj + 1]
                    port_sorted[pj + 1] = ptmp
                end if
                pj = pj + 1
            end while
            pi = pi + 1
        end while

        for port_entry in port_sorted
            out = out + NetworkMapper.format_port_line(port_entry) + char(10)
        end for
    end for

    return out
end function

    if not c.is_network_active then
        if not WifiManager.ensure_wifi() then return Error.init("No network connection.")
    end if

    data = NetworkMapper.scan(args["target"])
    if typeof(data) == "Error" then return data

    StateManager.cob["last_map"] = data
    return NetworkMapper.format(data)
end function

ExploitCommands.probe_cmd = function(args)

        if not StateManager.cob.hasIndex("active_addresses") or StateManager.cob["active_addresses"].len == 0 then

    catalog = ExploitManager.build_catalog(args["all"], not args["no_discover"], address, true)

ExploitCommands.overflow_cmd = function(args)
    lib = ExploitManager.get_active_lib()
    if typeof(lib) == "Error" then return lib

    address = args["address"]
    payload = args["payload"]
    opt = args["opt"]
    if opt == null then opt = ""

    result = lib.overflow(address, payload, opt)
    captured = ExploitManager.capture_result(address, payload, result)
    if captured != null then return captured

    out = char(10) + ("Overflow result").cbold("cyan") + char(10)
    out = out + ("  ").color("dim") + ExploitManager.format_result_type(result)
    return out

ExploitCommands.exploits_cmd = function(args)
    lib = ExploitManager.get_active_lib()
    if typeof(lib) == "Error" then return lib

    saved = ExploitManager.get_exploits(lib.lib_name, lib.version)
    out = char(10) + ("Cached exploits: ").cbold("cyan") + lib.lib_name.color("gold") + (" v").color("dim") + lib.version.color("dim") + char(10)

    if saved.len == 0 then
        return out + "  " + "(none saved yet - run scan to discover)".color("dim")
    end if

    for addr in saved.indexes
        addr_map = saved[addr]
        for payload in addr_map.indexes
            tuple = addr_map[payload]
            out = out + "  " + ExploitManager.format_access_label(tuple[1])
            if tuple[0] != "unknown" then out = out + (" as ").color("dim") + tuple[0].color("cyan")
            out = out + ("  ").color("dim") + payload.cbold("white") + char(10)
        end for
    end for
    out = out + char(10) + ("Run ").color("dim") + "scan".cbold("white") + (" to pick from this list.").color("dim")
    return out
end function

FileCommands = {}
FileCommands.classID = "FileCommands"

FileCommands.put = function(args)
    src = FileManager.resolve_path(args["source"])
    dest = FileManager.resolve_path(args["destination"])

    home_shell = StateManager.get_home_shell()
    if typeof(home_shell) == "Error" then return home_shell

    foc_shell = StateManager.get_focused_shell()
    if typeof(foc_shell) == "Error" then return foc_shell

    home_file = home_shell.host_computer.File(src)
    if home_file == null then return Error.init("Source not found: " + src)

    dest_folder = dest
    if dest_folder[-1:] != "/" then
        slash = dest_folder.lastIndexOf("/")
        if slash != null then dest_folder = dest_folder[:slash]
    end if

    result = home_shell.scp(src, dest_folder, foc_shell, true)
    if typeof(result) == "string" then return Error.init(result)
    return ("Uploaded ").color("green") + src.color("gold") + (" -> ").color("green") + dest.color("gold")
end function

FileCommands.get = function(args)
    src = FileManager.resolve_path(args["source"])
    dest = FileManager.resolve_path(args["destination"])

    home_shell = StateManager.get_home_shell()
    if typeof(home_shell) == "Error" then return home_shell

    foc_shell = StateManager.get_focused_shell()
    if typeof(foc_shell) == "Error" then return foc_shell

    result = foc_shell.scp(src, dest, home_shell, false)
    if typeof(result) == "string" then return Error.init(result)
    return ("Downloaded ").color("green") + src.color("gold") + (" -> ").color("green") + dest.color("gold")
end function

FileCommands.ls = function(args)
    path = current_path
    if args["path"] != null then path = FileManager.resolve_path(args["path"])

    c = StateManager.get_focused_computer()
    if typeof(c) == "Error" then return c

    folder = c.File(path)
    if folder == null then return Error.init("Path not found: " + path)
    if not folder.is_folder then return Error.init("Not a directory: " + path)

    result = path.cbold("cyan") + char(10)
    for f in folder.get_files
        name = f.name
        if f.is_folder then
            result = result + name.color("blue") + "/" + char(10)
        else if f.is_binary then
            result = result + name.color("gold") + char(10)
        else
            result = result + name.color("white") + char(10)
        end if
    end for
    return result
end function

FileCommands.cat = function(args)
    f = FileManager.get_file(args["path"])
    if typeof(f) == "Error" then return f
    if f.is_folder then return Error.init("Cannot cat a directory.")
    return f.get_content
end function

FileCommands.pwd = function(args)
    return current_path.color("gold")
end function

FileCommands.file_cmd = function(args)
    shell = StateManager.get_focused_shell()
    if typeof(shell) == "Error" then return shell

    computer = shell.host_computer
    explorer = FileManager.find_file_explorer(computer)
    if explorer == null then return Error.init("FileExplorer not found on this machine.")

    launch_path = explorer
    launch_args = ""

    if args["path"] != null then
        launch_args = FileManager.resolve_path(args["path"])
    end if

    if launch_args != "" then
        result = shell.launch(launch_path, launch_args)
    else
        result = shell.launch(launch_path)
    end if

    if typeof(result) == "string" then return Error.init(result)
    if result == 0 then return Error.init("Failed to launch FileExplorer (cooldown or permission issue).")

    out = ("Opened ").color("green") + explorer.cbold("white") + (" on ").color("green")
    out = out + StateManager.cob["focused_object"].info.local_ip.cbold("gold")
    if launch_args != "" then out = out + (" at ").color("dim") + launch_args.color("cyan")
    return out
end function

Magpie = {}

Magpie.build_prompt = function()
    obj = StateManager.cob["focused_object"]
    user = obj.info.user.cbold("cyan")
    local_ip = obj.info.local_ip.color("gold")
    path = current_path.color("green")
    return user + "@" + local_ip + ":" + path + "> "
end function

Magpie.print_result = function(result)
    if result == null then return
    if typeof(result) == "Error" then
        print(result.message.cbold("red"))
    else
        print(result)
    end if
end function

Magpie.run_repl = function()
    print("Magpie multitool loaded.".cbold("cyan"))
    print("Type ".color("dim") + "help".cbold("white") + " for commands, ".color("dim") + "exit".cbold("white") + " to quit.".color("dim"))
    print(("Workflow: ").color("dim") + "map".color("cyan") + " -> ".color("dim") + "connect".color("orange") + " -> ".color("dim") + "scan".color("green") + (" (pick shell/computer/file)").color("dim"))

    while true
        input = user_input(Magpie.build_prompt())
        if input == null then continue
        input = input.trim
        if input == "" then continue
        if input == "exit" or input == "quit" then break

        args = input.split(" ")
        result = MiniClick.run(args)
        Magpie.print_result(result)
    end while
end function
