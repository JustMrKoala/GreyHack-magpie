matches_criteria = function(item, key, value, negate = false)
    result = item[key] == value
    if negate then result = not result
    return result
end function

map.first = function(key, value)
    for each in self.indexes
        if matches_criteria(self[each], key, value) then return self[each]
    end for
end function
map.where = function(key, value)
    ret = {}
    for each in self.indexes
        if matches_criteria(self[each], key, value) then ret.push(self[each])
    end for
    return ret
end function
map.wherenot = function(key, value)
    ret = {}
    for each in self.indexes
        if matches_criteria(self[each], key, value, true) then ret.push(self[each])
    end for
    return ret
end function
map.hasMethod = function(method)
    ref = self
    if ref.hasIndex(method) then return true
    while ref.hasIndex("__isa")
        ref = ref["__isa"]
        if ref.hasIndex(method) then return true
    end while
    return false
end function
list.first = function(key, value)
    for each in self
        if typeof(each) == "string" then
            if key == "is" and each == value then return each
            if key == "contains" and each.is_match(value) then return each
            if key == "len" and each.len == value then return each
        else if typeof(each) == "file" then
            if key == "name" and each.name == value then return each
            if key == "namehas" and each.name.is_match(value) then return each
            if key == "path" and each.path == value then return each
            if key == "permissions" and each.permissions == value then return each
            if key == "has_permission" then
                v = value.values
                c = ""
                for p in v
                    if each.has_permission(p) then c = c + p
                end for
                if c == value then return each
            end if
            if key == "size" and each.size == value then return each
        else if typeof(each) == "port" then
            if key == "port_number" and each.port_number == value then return each
            if key == "is_closed" and each.is_closed == value then return each
            if key == "get_lan_ip" and each.get_lan_ip == value then return each
        else if typeof(each) == "map" then
            if each[key] == value then return each
        end if
    end for
end function

// CommandBuilder "Class"
CommandBuilder = function(name, description)
    obj = {}
    obj.name = name
    obj.description = description
    obj.options = {}
    obj.short_aliases = {} // short_name -> long_name
    obj.arguments = []

    obj.add_option = function(name, description, default = null, short_name = null, is_flag = false)
        option_data = {
            "name": name,
            "description": description,
            "default": default,
            "short_name": short_name,
            "is_flag": is_flag,
        }
        self.options[name] = option_data
        if short_name != null then
            self.short_aliases[short_name] = name
        end if
        return self // Enable method chaining
    end function

    obj.add_argument = function(name, description, is_rest = false)
        self.arguments.push({
            "name": name,
            "description": description,
            "is_rest": is_rest,
        })
        return self
    end function

    obj.set_function = function(func)
        self.func = @func
        return self
    end function

    obj.build = function()
        if MiniClick.commands.hasIndex(self.name) then
            print("Warning: Overwriting existing command '" + self.name + "'")
        end if
        MiniClick.commands[self.name] = {
            "name": self.name,
            "description": self.description,
            "options": self.options,
            "short_aliases": self.short_aliases,
            "arguments": self.arguments,
            "func": self["func"],
        }
    end function

    return obj
end function

// MiniClick Namespace Methods
MiniClick.create_command = function(name, description)
    return CommandBuilder(name, description)
end function

// Parsing Arguments and Options
MiniClick.parse_args = function(command, args)
    parsed = {}
    arg_index = 0
    i = 0

    handle_option = function(option_name, is_short)
        if not command.options.hasIndex(option_name) then
            return Error.init("Unknown option: " + option_name)
        end if
        option = command.options[option_name]
        if option.is_flag then
            parsed[option_name] = true
            return 1
        else
            if i + 1 >= args.len then
                return Error.init("Option " + (is_short and "-" or "--") + option_name + " requires a value")
            end if
            parsed[option_name] = args[i + 1]
            return 2
        end if
    end function

    while i < args.len
        arg = args[i]
        if arg[:2] == "--" then
            num_consumed = handle_option(arg[2:], false)
            if typeof(num_consumed) == "Error" then return num_consumed
            i = i + num_consumed
        else if arg[:1] == "-" then
            short_name = arg[1:]
            if not command.short_aliases.hasIndex(short_name) then
                return Error.init("Unknown option: " + short_name)
            end if
            long_name = command.short_aliases[short_name]
            num_consumed = handle_option(long_name, true)
            if typeof(num_consumed) == "Error" then return num_consumed
            i = i + num_consumed
        else
            if arg_index >= command.arguments.len then
                return Error.init("Too many arguments")
            end if
            arg_def = command.arguments[arg_index]
            arg_name = arg_def.name
            if arg_def.is_rest then
                parsed[arg_name] = args[i:]
                arg_index = arg_index + 1
                break
            else
                parsed[arg_name] = arg
                arg_index = arg_index + 1
                i = i + 1
            end if
        end if
    end while

    for option_name in command.options.indexes
        if not parsed.hasIndex(option_name) then
            if command.options[option_name].is_flag then
                parsed[option_name] = false
            else
                parsed[option_name] = command.options[option_name].default
            end if
        end if
    end for

    if arg_index < command.arguments.len then
        return Error.init("Not enough arguments")
    end if

    return parsed
end function

// Generating Help Text
MiniClick.generate_help = function(command_name = null)
    if command_name == null then
        help_text = "Available commands:\n"
        for name in MiniClick.commands.indexes
            command = MiniClick.commands[name]
            help_text = help_text + name + ": " + command.description + "\n"
        end for
        if MiniClick.aliases.indexes.len > 0 then
            help_text = help_text + "\nAliases:\n"
            for alias in MiniClick.aliases.indexes
                help_text = help_text + "  " + alias + " -> " + MiniClick.aliases[alias] + "\n"
            end for
        end if
        help_text = help_text + "\nUse 'command --help' for more information on a specific command."
    else
        if MiniClick.aliases.hasIndex(command_name) then
            command_name = MiniClick.aliases[command_name]
        end if
        if not MiniClick.commands.hasIndex(command_name) then
            return Error.init("Unknown command: " + command_name)
        end if
        command = MiniClick.commands[command_name]
        help_text = command_name + ": " + command.description + "\n\nArguments:\n"
        for arg in command.arguments
            help_text = help_text + "  " + arg.name + ": " + arg.description + "\n"
        end for

        help_text = help_text + "\nOptions:\n"
        for option_name in command.options.indexes
            option = command.options[option_name]
            help_text = help_text + "  --" + option.name
            if option.short_name != null then
                help_text = help_text + " (-" + option.short_name + ")"
            end if
            help_text = help_text + ": " + option.description
            if option.is_flag then
                help_text = help_text + " (flag)"
            else if option.default != null then
                help_text = help_text + " (default: " + option.default + ")"
            end if
            help_text = help_text + "\n"
        end for
    end if
    return help_text
end function

// Running Commands
MiniClick.run = function(args)
    command_name = args[0]
    if MiniClick.aliases.hasIndex(command_name) then
        command_name = MiniClick.aliases[command_name]
    end if
    if not MiniClick.commands.hasIndex(command_name) then
        return Error.init("Unknown command: " + command_name)
    end if

    command = MiniClick.commands[command_name]

    // --help may appear anywhere in the argument list
    for a in args[1:]
        if a == "--help" then return MiniClick.generate_help(command_name)
    end for

    parsed_args = MiniClick.parse_args(command, args[1:])
    if parsed_args == null or typeof(parsed_args) == "Error" then return parsed_args

    if command["func"] == null then
        return Error.init("Error: No function defined for command '" + command_name + "'")
    end if

    result = command.func(parsed_args)
    if result != null then
        return result
    end if
end function

// Checks random IP addresses for an open port 1542 and returns the IP address.
// @return {string} - The IP address of the hackshop, or null if not found.
get_hackshop = function()
    max_attempts = 50
    attempts = 0
    
    while attempts < max_attempts
        ip = get_random_ip()
        
        if not is_lan_ip(ip) then
            router = get_router(ip)
            
            if router != null then
                ports = router.used_ports
                
                for port in ports
                    if port.port_number == 1542 and not port.is_closed then
                        return router.public_ip
                    end if
                end for
            end if
        end if
        
        attempts = attempts + 1
    end while
    
    return null
end function

StateManager.load_plugins = function()
    for plugin_name in self.cob["plugin_stores"].indexes
        p_data = self.cob["plugin_stores"][plugin_name]

        for cmd in p_data.commands
            builder = MiniClick.create_command(cmd.name, cmd.description)

            if cmd.hasIndex("arguments") then
                for arg in cmd.arguments
                    builder.add_argument(arg.name, arg.description, arg.hasIndex("is_rest") and arg.is_rest)
                end for
            end if

            if cmd.hasIndex("options") then
                for opt in cmd.options.indexes
                    opt_data = cmd.options[opt]
                    builder.add_option(opt_data.name, opt_data.description, opt_data.default, opt_data.short_name, opt_data.is_flag)
                end for
            end if

            builder.set_function(cmd["function"]).build()
        end for
    end for

    print(self.cob["plugin_stores"].len + " plugins loaded.")
end function

StateCommands = {} // CLI commands for interacting with StateManager

StateCommands.list_objects = function(args)
    result = ""
    for obj in StateManager.cob["stored_objects"] // Print UID, type, and IP addresses
        key = obj["uid"]
        result = result + key + ":"

        if StateManager.cob["focused_object"]["uid"] == obj["uid"] then result = result + " (focused)"
        if StateManager["relative_object"]["uid"] == obj["uid"] then result = result + " (relative)"
        if StateManager.cob["home_object"]["uid"] == obj["uid"] then result = result + " (home)"

        result = result + " " + obj["info"]["user"]

        if obj["shell"] != null then
            result = result + " shell "
        else if obj["computer"] != null then
            result = result + " computer "
        else 
            result = result + " file "
        end if

        result = result + obj.info["local_ip"] + "@" + obj.info["pub_ip"] + char(10)
    end for
    return result
end function

StateCommands.focus_object = function(args)
    uid = args["uid"]

    obj = StateManager.cob["stored_objects"].first("uid", uid)
    if obj != null then
        StateManager.cob["focused_object"] = obj
        return
    end if

    return Error.init("Unknown object: " + uid)
end function

StateCommands.release_object = function(args)
    uid = args["uid"]
    obj = StateManager.cob["stored_objects"].first("uid", uid)

    if obj == null then
        return Error.init("Unknown object: " + uid)
    end if

    if StateManager.cob["focused_object"]["uid"] == uid then return Error.init("Cannot release focused object.")
    if StateManager.cob["relative_object"]["uid"] == uid then return Error.init("Cannot release relative object.")
    if StateManager.cob["home_object"]["uid"] == uid then return Error.init("Cannot release home object.")

    index = StateManager.cob["stored_objects"].indexOf(obj)

    confirm = user_input("Are you sure you want to release this object? (y/n): ")
    if confirm != "y" then return

    return StateManager.cob["stored_objects"].remove(index)
end function

UtilityCommands.set_alias = function(args)
    alias = args["alias"]
    command = args["command"]

    if MiniClick.commands.hasIndex(alias) then
        return Error.init("Name already used by a command: " + alias)
    end if

    if MiniClick.aliases.hasIndex(alias) then
        return Error.init("Alias already exists: " + alias)
    end if

    // Resolve through aliases so chained aliases point at the real command
    if MiniClick.aliases.hasIndex(command) then
        command = MiniClick.aliases[command]
    end if

    if not MiniClick.commands.hasIndex(command) then
        return Error.init("Unknown command: " + command)
    end if

    MiniClick.aliases[alias] = command
end function

UtilityCommands.unset_alias = function(args)
    alias = args["alias"]

    if not MiniClick.aliases.hasIndex(alias) then
        return Error.init("Unknown alias: " + alias)
    end if

    MiniClick.aliases.remove(alias)
    return "Alias removed: " + alias
end function

    hash = args["hash"]
    pass = StateManager.cob["home_crypto"].decipher(hash)
    if pass == null then return Error.init("Hash not found.") else return pass
end function

ExploitManager = {}
ExploitManager.classID = "ExploitManager"

ExploitManager.whois_lookup = function(ip)
    info = ""

    if not is_valid_ip(ip) and nslookup(ip) == "Not found" then return Error.init("Invalid IP address or domain: " + ip)

    if not is_valid_ip(ip) then
        info = info + "Domain: " + ip + "\n"
        ip = nslookup(ip)
    end if

    info = info + "IP: " + ip + "\n" + whois(ip)
    return info
end function

ExploitManager.smtp_enum = function(ip, port)
    port = port.to_int
    if typeof(port) != "number" then return Error.init("Invalid port: " + port)
    if not is_valid_ip(ip) then return Error.init("Invalid IP address: " + ip)

    result = StateManager.cob["home_crypto"].smtp_user_list(ip, port)

    if typeof(result) == "string" then return Error.init("Enumeration failed: " + result)

    return result
end function

ExploitManager.full_enum = function(ip)
    if not is_valid_ip(ip) then return Error.init("Invalid IP address: " + ip)

    result = ""

    router = get_router(ip)
    if router == null then return Error.init("No router found for IP: " + ip)

    result = result + "Router LAN IP: " + router.local_ip + "\n"
    result = result + "Router BSSID: " + router.bssid_name + "\n"
    result = result + "Router ESSID: " + router.essid_name + "\n"
    result = result + "Router kernel: " + router.kernel_version + "\n\n"
    result = result + "Port forwards detected:\n"

    for port in router.used_ports
        result = result + port.port_number + "  open:" + (not port.is_closed) + "  " + router.port_info(port) + "  " + port.get_lan_ip + "\n"
    end for

    result = result + "\nFirewall rules:\n"

    for rule in router.firewall_rules
        result = result + rule + "\n"
    end for

    result = result + "Connected devices:\n"

    for lan in router.devices_lan_ip
        result = result + "\nLAN: " + lan + "\n"
        result = result + "Ports:\n"

        ports = router.device_ports(lan)

        for port in ports
            result = result + port.port_number + "  open:" + (not port.is_closed) + "  " + router.port_info(port) + " " + port.get_lan_ip + "\n"
        end for
    end for

    return result
end function

ExploitManager.remote_lib = function(ip, port)
    port = port.to_int
    if typeof(port) != "number" then return Error.init("Invalid port: " + port)
    session = StateManager.cob["remote_metax"].net_use(ip, port)
    if session == null then return Error.init("net_use failed for " + ip + ":" + port)
    return session.dump_lib
end function

ExploitManager.local_lib = function(path)
    return StateManager.cob["remote_metax"].load(path)
end function

ExploitManager.get_addresses = function(meta_lib)
    return StateManager.cob["home_metax"].scan(meta_lib)
end function

ExploitManager.parse_address = function(unparsed)
    segments = unparsed.split("Unsafe check: ")[1:]
    
    parsed = []
    for exploit in segments
        match = exploit.matches("<b>\S*<\/b>\.")
        parsed.push(match.values.pull[3:match.len-6])
    end for

    return parsed
end function

ExploitManager.verify_tuple = function(tuple)
    if tuple.len != 2 then return false
    if indexOf(["root", "user", "guest", "unknown"], tuple[0]) == null then return false
    if indexOf(["shell", "computer", "file", "number"], tuple[1]) == null then return false
    return true
end function

ExploitManager.save_exploit = function(libname, libversion, address, string, result_tuple)
    if not self.verify_tuple(result_tuple) then return Error.init("Invalid tuple: " + result_tuple)
    
    if not StateManager.cob["exploit_store"].hasIndex(libname) then
        StateManager.cob["exploit_store"][libname] = {}
    end if

    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then
        StateManager.cob["exploit_store"][libname][libversion] = {}
    end if

    if not StateManager.cob["exploit_store"][libname][libversion].hasIndex(address) then
        StateManager.cob["exploit_store"][libname][libversion][address] = {}
    end if

    StateManager.cob["exploit_store"][libname][libversion][address][string] = result_tuple
end function

ExploitManager.get_exploits = function(libname, libversion)
    if not StateManager.cob["exploit_store"].hasIndex(libname) then return {}
    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then return {}

    return StateManager.cob["exploit_store"][libname][libversion]
end function

// Returns whether the library directory exists, and creates it if it doesn't
ExploitManager.check_libpath = function(libname, libversion)
    c = StateManager.get_home_computer()
    if typeof(c) == "Error" then return c

    home_path = StateManager.cob["home_path"]

    exploits_path = home_path + "/exploits"
    lib_path = exploits_path + "/" + libname
    version_path = lib_path + "/" + libversion

    // Function to create a directory if it doesn't exist
    check_dir = function(path, name)
        if not c.File(path + "/" + name) then
            c.create_folder(path, name)
            return false
        end if
        return true
    end function

    // Ensure the necessary directories exist
    check_dir(home_path, "exploits")
    check_dir(exploits_path, libname)
    check_dir(lib_path, libversion)

    // Return whether the version directory has files
    return c.File(version_path).get_files.len != 0
end function

ExploitManager.check_exploit = function(libname, libversion, address = null)
    if not StateManager.cob["exploit_store"].hasIndex(libname) then return false
    if not StateManager.cob["exploit_store"][libname].hasIndex(libversion) then return false
    if address != null and not StateManager.cob["exploit_store"][libname][libversion].hasIndex(address) then return false

    return true
end function

ExploitManager.load_exploit = function(libname, libversion)
    c = StateManager.get_home_computer()
    if typeof(c) == "Error" then return c

    self.check_libpath(libname, libversion)
    version_path = StateManager.cob["home_path"] + "/exploits/" + libname + "/" + libversion

    version_dir = c.File(version_path)
    if version_dir == null or version_dir.is_folder == false then
        // No directory or not a folder
        return
    end if

    exploits = {}
    for file in version_dir.get_files
        if file.is_folder then continue
        exploits[file.name] = file
    end for

    return exploits
end function
